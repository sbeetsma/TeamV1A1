DROP TABLE IF EXISTS Sessions;
DROP TABLE IF EXISTS Categories;
DROP TABLE IF EXISTS Products;
DROP TABLE IF EXISTS Ordered_products;



CREATE TABLE Sessions
(
	session_id VARCHAR(128) NOT NULL,
	segment VARCHAR(255) NOT NULL,
	PRIMARY KEY(session_id)
);

CREATE TABLE Categories
(
	category_id INT NOT NULL,
	category_name VARCHAR(255) NOT NULL,
	PRIMARY KEY(category_id),
);

CREATE TABLE Products
(
	product_id VARCHAR(128) NOT NULL,
	product_name VARCHAR(255) NOT NULL,
	category INT,
	sub_category INT,
	sub_sub_category INT,
	sub_sub_sub_category INT,
	PRIMARY KEY(product_id),
	FOREIGN KEY(category) REFERENCES Category(category_id),
	FOREIGN KEY(sub_category) REFERENCES Category(category_id),
	FOREIGN KEY(sub_sub_lowest_category) REFERENCES Category(category_id),
	FOREIGN KEY(sub_sub_sub_lowest_category) REFERENCES Category(category_id),
);
CREATE TABLE Ordered_products
(
	session_id VARCHAR(128) NOT NULL,
	product_id VARCHAR(128) NOT NULL,
	quantity INT NOT NULL,
	PRIMARY KEY(session_id, product_id),
	FOREIGN KEY(session_id) REFERENCES Session(session_id),
	FOREIGN KEY(product_id) REFERENCES Product(product_id)
);
